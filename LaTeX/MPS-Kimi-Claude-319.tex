	\section{Data Structures}
	
	An MPS with open boundary conditions is stored as a list of 3-index tensors:
	\begin{itemize}
		\item $A^{(i)}_{a_{i-1},\sigma_i,a_i}$: Left-canonical tensor at site $i$,
		dimensions $(D_{i-1} \times d \times D_i)$
		\item $B^{(i)}_{a_{i-1},\sigma_i,a_i}$: Right-canonical tensor,
		dimensions $(D_{i-1} \times d \times D_i)$
		\item $M^{(i)}_{a_{i-1},\sigma_i,a_i}$: General (non-canonical) tensor
		during optimization
		\item Physical dimension $d$ (e.g., $d=2$ for spin-1/2, $d=3$ for spin-1)
	\end{itemize}
	
	\textit{Storage Convention}: For site $i$, tensor $T^{(i)}$ has shape
	$(D_{i-1}, d, D_i)$ where the middle index is physical.
	
	\textit{Matrix Reshape Convention (Critical)}: When flattening
	$T^{(i)}_{a_{i-1},\sigma_i,a_i}$ to matrix $\tilde{M}_{r,c}$ for QR/SVD,
	the row index $r$ must follow C-order (row-major) mapping:
	\begin{equation}
		r = a_{i-1} \cdot d + \sigma_i \quad
		\text{where } a_{i-1} \in [0,D_{i-1}-1],\; \sigma_i \in [0,d-1]
	\end{equation}
	Column index $c = a_i$ (right bond).
	This yields matrix shape $(D_{i-1}d) \times D_i$.
	
	\textit{Python/NumPy Implementation}:
	\begin{verbatim}
		# C-order (row-major) reshape - CRITICAL for index consistency
		M_tilde = M.reshape((D_left * d, D_right), order='C')
		# Verify: M_tilde[a_prev * d + sigma, a_curr] == M[a_prev, sigma, a_curr]
	\end{verbatim}
	\textit{Fortran/MATLAB}: Use \texttt{reshape(..., order='F')} with reversed
	index ordering $r = \sigma_i \cdot D_{i-1} + a_{i-1}$ (column-major), or
	transpose after C-order reshape.

	\section{Canonicalization Algorithms}
	
	\subsection{Left-Canonicalization (QR-based)}
	More stable than SVD for canonicalization when singular values aren't needed.
	
	\begin{algorithm}[H]
		\caption{LeftCanonicalize (Numerically Stable)}
		\begin{algorithmic}[1]
			\Require MPS tensors $\{M^{(1)}, \dots, M^{(L)}\}$ with shapes $(D_{i-1}, d, D_i)$
			\Ensure Left-canonical $\{A^{(1)}, \dots, A^{(L)}\}$, last tensor $C$
			\State $C \gets 1.0$ \Comment{Initial transfer matrix, $1 \times 1$}
			\For{$i = 1$ to $L-1$}
			\State $\tilde{M}_{(a_{i-1}\sigma_i), a_i} \gets
			\sum_{a'_{i-1}} C_{a'_{i-1}, a_{i-1}} M^{(i)}_{a'_{i-1}, \sigma_i, a_i}$
			\State \Comment{Matrix shape: $(D_{i-1}d) \times D_i$}
			\If{$D_{i-1} \cdot d < D_i$}
			\State \textbf{Warning:} Bond dimension exceeds physical capacity.
			\EndIf
			\State $Q, R \gets \text{QR}_{\text{econ}}(\tilde{M})$
			\If{$D_{i-1} \cdot d \geq D_i$}
			\State $A^{(i)} \gets \text{reshape}(Q[:, :D_i], (D_{i-1}, d, D_i))$
			\State $C \gets R$
			\Else
			\State $A^{(i)} \gets \text{reshape}(Q, (D_{i-1}, d, D_{i-1}d))$
			\State $C \gets R$
			\EndIf
			\EndFor
			\State $A^{(L)} \gets \sum_{a'_{L-1}} C_{a'_{L-1}, a_{L-1}}
			M^{(L)}_{a'_{L-1}, \sigma_L, 1}$
		\end{algorithmic}
	\end{algorithm}
	
	
	\subsection{Right-Canonicalization (RQ-based)}
	
	\begin{algorithm}[H]
		\caption{RightCanonicalize (Numerically Stable)}
		\begin{algorithmic}[1]
			\Require MPS tensors $\{M^{(1)}, \dots, M^{(L)}\}$ with shapes $(D_{i-1}, d, D_i)$
			\Ensure Right-canonical $\{B^{(1)}, \dots, B^{(L)}\}$
			\State $C \gets 1.0$ \Comment{Initial transfer matrix, $1 \times 1$}
			\For{$i = L$ down to $2$}
			\State $\tilde{M}_{a_{i-1}, (\sigma_i a_i)} \gets
			\sum_{a'_i} M^{(i)}_{a_{i-1}, \sigma_i, a'_i} C_{a'_i, a_i}$
			\State \Comment{Matrix shape: $D_{i-1} \times (d \cdot D_i)$}
			\State $Q, R \gets \text{QR}(\tilde{M}^T)$;
			$R \gets R^T$, $Q \gets Q^T$
			\State $B^{(i)} \gets \text{reshape}(Q, (D_{i-1}, d, D_i))$
			\State $C \gets R$
			\EndFor
			\State $B^{(1)} \gets \sum_{a'_1} C_{1, a'_1} M^{(1)}_{1, \sigma_1, a'_1}$
		\end{algorithmic}
	\end{algorithm}
	
	\textit{Python Implementation Note:}
	\begin{verbatim}
		# Right-canonicalize site i (QR of transpose)
		M_tilde = M.reshape((D_left, d * D_right))
		Q, R = np.linalg.qr(M_tilde.T)   # Q: (d*D_right, k), R: (k, D_left)
		k = Q.shape[1]
		B = Q.T.reshape(k, d, D_right)    # new right-canonical tensor
		M_prev = np.einsum('ijk,kl->ijl', M_prev, R.T)  # absorb into left neighbor
	\end{verbatim}
	
	\subsection{Mixed-Canonical Form at Bond $k$}
	Required for variational minimization.
	
	\begin{algorithm}[H]
		\caption{BringToBond}
		\begin{algorithmic}[1]
			\Require MPS $\{M^{(i)}\}$, target bond $k$
			\Ensure Left-canonical $A^{(1..k)}$, right-canonical $B^{(k+1..L)}$
			\State Left-canonicalize sites $1 \dots k$ using QR sweeps
			\State Right-canonicalize sites $L \dots k+1$ using RQ sweeps
		\end{algorithmic}
	\end{algorithm}

	\section{Truncation and Compression}
	
	\subsection{SVD Compression with Tolerance}
	\begin{algorithm}[H]
		\caption{TruncateBond}
		\begin{algorithmic}[1]
			\Require Matrix $M$ ($m \times n$), max bond $D_{\max}$,
			tolerance $\eps$, abs.\ tolerance $\eps_{\text{abs}} \sim 10^{-14}$
			\Ensure Truncated $U, S, V^\dagger$ with $D \leq D_{\max}$
			\State $U, S, V^\dagger \gets \text{SVD}(M, \text{full\_matrices=False})$
			\State $s_i \gets \text{diag}(S)$ in descending order
			\If{$s_0 < \eps_{\text{abs}}$}
			\State Set $D \gets 1$
			\Else
			\State $\text{threshold} \gets \max(\eps \cdot s_0,\; \eps_{\text{abs}})$
			\State $D \gets \max(1,\;\min(D_{\max},\;\text{count}(s_i > \text{threshold})))$
			\EndIf
			\State \textbf{return} $U[:, :D]$, $S[:D]$, $V^\dagger[:D, :]$
		\end{algorithmic}
	\end{algorithm}
	
	\textit{Rationale}: Absolute tolerance prevents discarding states due to
	global rescaling. For normalized states, $\eps_{\text{abs}} = 10^{-14}$
	captures the machine precision floor.
	
	\subsection{Variational Compression (Two-Site)}
	\begin{algorithm}[H]
		\caption{VariationalCompress}
		\begin{algorithmic}[1]
			\Require MPS $\{\psi\}$ with bond dimensions $D_{\text{old}}$,
			target $D_{\text{new}} < D_{\text{old}}$
			\Ensure Compressed MPS $\{\phi\}$ minimizing $\|\psi - \phi\|^2$
			\For{sweep $= 1$ to $N_{\text{sweeps}}$}
			\For{$i = 1$ to $L-1$}
			\State Contract $\psi^{(i)}\psi^{(i+1)}$ to form two-site tensor $\Theta$
			\State SVD: $\Theta = U S V^\dagger$ with truncation to $D_{\text{new}}$
			\State Update: $\phi^{(i)} \gets U$, $\phi^{(i+1)} \gets S V^\dagger$
			\EndFor
			\State Repeat right-to-left
			\EndFor
		\end{algorithmic}
	\end{algorithm}

	\section{DMRG Algorithm}
	
	\subsection{MPO Convention}
	
	Throughout this document we adopt the convention
	\begin{equation}
		W^{(i)}[a_{\text{in}},\, a_{\text{out}},\, \sigma,\, \sigma']
	\end{equation}
	where $a_{\text{in}}$ ($a_{\text{out}}$) is the left (right) MPO bond index
	and $\sigma$ ($\sigma'$) is the ket (bra) physical index.
	The full Hamiltonian is assembled as
	\begin{equation}
		H = W^{(1)}[a_0, a_1]\; W^{(2)}[a_1, a_2]\;\cdots\; W^{(L)}[a_{L-1}, a_L]
	\end{equation}
	contracting over all shared bond indices $a_1, \dots, a_{L-1}$.
	
	\subsection{Environment Tensors}
	
	\textbf{Left Environment}
	$\mathcal{L}^{(i)}[a,\, x,\, u]$: contraction of MPS, MPO, and conjugate MPS
	from site $1$ to $i-1$.
	Index sizes: $a \in [0, D_W-1]$, $x, u \in [0, D_{i-1}-1]$.
	
	\textbf{Right Environment}
	$\mathcal{R}^{(i)}[a,\, z,\, v]$: contraction from site $i$ to $L$.
	Index sizes: $a \in [0, D_W-1]$, $z, v \in [0, D_i-1]$.
	
	\textbf{Boundary conditions}:
	\begin{align}
		\mathcal{L}^{(1)}[a, 1, 1] &= 1\quad (\text{scalar }1\times1\times1\text{ tensor})\\
		\mathcal{R}^{(L+1)}[a, 1, 1] &= 1\quad (\text{scalar }1\times1\times1\text{ tensor})
	\end{align}
	
	\subsection{Environment Update Formulas}
	
	\textbf{Left Environment Update} (absorb site $i$, sweep left-to-right):
	\begin{equation}
		\mathcal{L}^{(i+1)}[B, z, v]
		= \sum_{a,x,u,\,\sigma,\sigma'}
		\mathcal{L}^{(i)}[a, x, u]\cdot
		M^{(i)}[x, \sigma, z]\cdot
		W^{(i)}[a, B, \sigma, \sigma']\cdot
		\overline{M^{(i)}}[u, \sigma', v]
	\end{equation}
	In einsum notation:
	\begin{equation}
		\boxed{
			\mathcal{L}^{(i+1)}
			= \texttt{einsum('axu,\,xsz,\,aBsS,\,uSv\,->\,Bzv',}\;
			\mathcal{L}^{(i)},\, M,\, W,\, M^*\texttt{)}
		}
	\end{equation}
	
	\begin{algorithm}[H]
		\caption{UpdateLeftEnvironment}
		\begin{algorithmic}[1]
			\Require $\mathcal{L}^{(i)}$ shape $(D_W, D_{i-1}, D_{i-1})$,\;
			$M^{(i)}$ shape $(D_{i-1}, d, D_i)$,\;
			$W^{(i)}$ shape $(D_W, D_W, d, d)$
			\Ensure $\mathcal{L}^{(i+1)}$ shape $(D_W, D_i, D_i)$
			\State $\mathcal{L}^{(i+1)} \gets
			\texttt{einsum('axu,xsz,aBsS,uSv->Bzv'},\;
			\mathcal{L}^{(i)},\, M^{(i)},\, W^{(i)},\, \overline{M^{(i)}}\texttt{)}$
			\Comment{$a$=$a_{\rm in}$, $B$=$a_{\rm out}$, $x,u$=left MPS bonds,
				$z,v$=right MPS bonds, $s$=ket, $S$=bra physical}
		\end{algorithmic}
	\end{algorithm}
	
	\textbf{Right Environment Update} (absorb site $i$, sweep right-to-left):
	\begin{equation}
		\mathcal{R}^{(i)}[a, x, u]
		= \sum_{B,z,v,\,\sigma,\sigma'}
		\mathcal{R}^{(i+1)}[B, z, v]\cdot
		M^{(i)}[x, \sigma, z]\cdot
		W^{(i)}[a, B, \sigma, \sigma']\cdot
		\overline{M^{(i)}}[u, \sigma', v]
	\end{equation}
	In einsum notation:
	\begin{equation}
		\boxed{
			\mathcal{R}^{(i)}
			= \texttt{einsum('Bzv,\,xsz,\,aBsS,\,uSv\,->\,axu',}\;
			\mathcal{R}^{(i+1)},\, M,\, W,\, M^*\texttt{)}
		}
	\end{equation}
	
	\begin{algorithm}[H]
		\caption{UpdateRightEnvironment}
		\begin{algorithmic}[1]
			\Require $\mathcal{R}^{(i+1)}$ shape $(D_W, D_i, D_i)$,\;
			$M^{(i)}$ shape $(D_{i-1}, d, D_i)$,\;
			$W^{(i)}$ shape $(D_W, D_W, d, d)$
			\Ensure $\mathcal{R}^{(i)}$ shape $(D_W, D_{i-1}, D_{i-1})$
			\State $\mathcal{R}^{(i)} \gets
			\texttt{einsum('Bzv,xsz,aBsS,uSv->axu',}\;
			\mathcal{R}^{(i+1)},\, M^{(i)},\, W^{(i)},\, \overline{M^{(i)}}\texttt{)}$
			\Comment{$B$=$a_{\rm in}$(right), $a$=$a_{\rm out}$(left),
				$z,v$=right MPS bonds, $x,u$=left MPS bonds}
		\end{algorithmic}
	\end{algorithm}
	
	\textit{Key consistency check}: \texttt{UpdateLeftEnvironment} and
	\texttt{UpdateRightEnvironment} are exact transposes of each other:
	swapping $(B\leftrightarrow a)$ and $(z,v\leftrightarrow x,u)$ in the
	left update gives the right update.
	
	\subsection{Two-Site Effective Hamiltonian}
	
	For two-site optimization at sites $i$ and $j=i+1$, with
	$\Theta[x, s_i, s_j, z]$ the two-site tensor:
	\begin{multline}
		(\mathcal{H}_{\rm eff}\,\Theta)[u, S_i, S_j, v]
		= \sum_{a, b, x, z,\, s_i, s_j}
		\mathcal{L}^{(i)}[a,x,u]\cdot
		\Theta[x, s_i, s_j, z]\cdot
		W^{(i)}[a, b, s_i, S_i]\cdot
		W^{(j)}[b, B, s_j, S_j]\cdot
		\mathcal{R}^{(j+1)}[B,z,v]
	\end{multline}
	Implemented as a two-step contraction to avoid $O(D^4)$ intermediate tensors:
	\begin{align}
		T[u, b, S_i, s_j, z]
		&= \texttt{einsum('axu,\,xisz,\,abiS\,->\,ubSsz',}\;
		\mathcal{L}^{(i)},\,\Theta,\, W^{(i)}\texttt{)}
		\label{eq:matvec_step1}\\
		(\mathcal{H}_{\rm eff}\,\Theta)[u, S_i, S_j, v]
		&= \texttt{einsum('ubSsz,\,bBsT,\,Bzv\,->\,uSTv',}\;
		T,\, W^{(j)},\, \mathcal{R}^{(j+1)}\texttt{)}
		\label{eq:matvec_step2}
	\end{align}
	
	\begin{algorithm}[H]
		\caption{TwoSiteMatvec}
		\begin{algorithmic}[1]
			\Require $\mathcal{L}^{(i)}$ $(D_W, D_l, D_l)$,\;
			$\mathcal{R}^{(j+1)}$ $(D_W, D_r, D_r)$,\;
			$W^{(i)}$ $(D_W, D_W, d, d)$,\;
			$W^{(j)}$ $(D_W, D_W, d, d)$,\;
			$\Theta$ reshaped from flat vector $(D_l d^2 D_r)$
			\Ensure $\mathcal{H}_{\rm eff}\,\Theta$ as flat vector
			\State $\Theta \gets v.\text{reshape}(D_l, d, d, D_r)$
			\State $T \gets \texttt{einsum('axu,xisz,abiS->ubSsz',}\;
			\mathcal{L}^{(i)},\,\Theta,\,W^{(i)}\texttt{)}$
			\State $\text{out} \gets \texttt{einsum('ubSsz,bBsT,Bzv->uSTv',}\;
			T,\,W^{(j)},\,\mathcal{R}^{(j+1)}\texttt{)}$
			\State \textbf{return} $\text{out}.\text{reshape}(-1)$
		\end{algorithmic}
	\end{algorithm}
	
	\subsection{Local Optimization}
	
	\begin{algorithm}[H]
		\caption{OptimizeTwoSite}
		\begin{algorithmic}[1]
			\Require Site indices $i, j=i+1$; environments
			$\mathcal{L}^{(i)}, \mathcal{R}^{(j+1)}$;
			MPO tensors $W^{(i)}, W^{(j)}$;
			current tensors $M^{(i)}, M^{(j)}$; $D_{\max}$
			\Ensure Updated tensors $M^{(i)}_{\rm new}, M^{(j)}_{\rm new}$; energy $E_0$
			\State $\Theta_0 \gets \texttt{einsum('ijk,klm->ijlm',}\, M^{(i)}, M^{(j)}\texttt{)}$
			\Comment{Shape $(D_l, d, d, D_r)$}
			\State Define \texttt{matvec} using Eqs.~\eqref{eq:matvec_step1}--\eqref{eq:matvec_step2}
			\State $E_0, \Theta_{\rm opt} \gets
			\texttt{eigsh}(\mathcal{H}_{\rm eff},\, k=1,\, \text{which='SA'},\,
			v_0=\Theta_0.\text{reshape}(-1))$
			\State $U, s, V^\dagger \gets \text{SVD}(\Theta_{\rm opt}.\text{reshape}(D_l d,\, d D_r))$
			\Comment{Truncate to $D_{\max}$}
			\State $M^{(i)}_{\rm new} \gets U.\text{reshape}(D_l, d, D_{\rm new})$
			\State $M^{(j)}_{\rm new} \gets (\text{diag}(s)\cdot V^\dagger).\text{reshape}(D_{\rm new}, d, D_r)$
			\State \textbf{return} $M^{(i)}_{\rm new},\, M^{(j)}_{\rm new},\, E_0$
		\end{algorithmic}
	\end{algorithm}
	
	\subsection{Main Two-Site DMRG Loop}
	
	\begin{algorithm}[H]
		\caption{DMRG\_TwoSite}
		\begin{algorithmic}[1]
			\Require Initial MPS, MPO $\{W\}$, $D_{\max}$, $N_{\rm sweep}$, tolerance $\eps_E$
			\Ensure Ground state MPS, energy $E_0$
			\State Right-canonicalize full MPS
			\State Build all right environments: $\mathcal{R}^{(L)}, \dots, \mathcal{R}^{(1)}$
			\State $\mathcal{L}^{(1)} \gets \mathbf{1}_{1\times1\times1}$
			\For{sweep $= 1$ to $N_{\rm sweep}$}
			\State \Comment{Left-to-right pass}
			\For{$i = 0$ to $L-2$}
			\State $M^{(i)}, M^{(i+1)}, E \gets \texttt{OptimizeTwoSite}(i, i+1)$
			\State Left-canonicalize $M^{(i)}$ via QR: $M^{(i)} = Q$, absorb $R$ into $M^{(i+1)}$
			\State $\mathcal{L}^{(i+1)} \gets \texttt{UpdateLeftEnvironment}(\mathcal{L}^{(i)}, i)$
			\EndFor
			\State \Comment{Right-to-left pass}
			\For{$i = L-2$ down to $0$}
			\State $M^{(i)}, M^{(i+1)}, E \gets \texttt{OptimizeTwoSite}(i, i+1)$
			\State Right-canonicalize $M^{(i+1)}$ via RQ: absorb $R$ into $M^{(i)}$
			\State $\mathcal{R}^{(i+1)} \gets \texttt{UpdateRightEnvironment}(\mathcal{R}^{(i+2)}, i+1)$
			\EndFor
			\If{$|E - E_{\rm prev}| < \eps_E$ \textbf{and} sweep $\geq 3$}
			\State \textbf{break} \Comment{Converged}
			\EndIf
			\State $E_{\rm prev} \gets E$
			\EndFor
			\State \textbf{return} MPS, $E$
		\end{algorithmic}
	\end{algorithm}
	
	\subsection{Observables and Measurements}
	
	\subsubsection{Transfer Matrix Convention}
	
	All single-layer transfer matrices use the verified einsum convention:
	\begin{align}
		\text{No operator:}&\quad
		\texttt{einsum('xy,xsz,ysw->zw',}\;\text{env},\,M,\,M^*\texttt{)}
		\label{eq:transfer_no_op}\\
		\text{With operator }O:&\quad
		\texttt{einsum('xy,xsz,st,ytw->zw',}\;\text{env},\,M,\,O,\,M^*\texttt{)}
		\label{eq:transfer_with_op}
	\end{align}
	Index assignment: \texttt{env}$[x,y]$, $M[x,s,z]$, $M^*[y,t,w]$
	where $s,t$ are ket/bra physical indices and $z,w$ are ket/bra right bonds.
	Using distinct letters for all three indices of $M^*$ is critical to avoid
	index collisions.
	
	\subsubsection{Local Expectation Values}
	
	\begin{equation}
		\langle O^{(i)} \rangle
		= \sum_{\text{all sites}} \prod_{k\neq i} (\text{identity transfer})
		\times (\text{operator transfer at }i)
	\end{equation}
	\begin{verbatim}
		def expectation(mps_obj, op, site):
		env = np.ones((1, 1))
		for i in range(mps_obj.L):
		M = mps_obj.tensors[i]
		if i == site:
		env = np.einsum('xy,xsz,st,ytw->zw', env, M, op, M.conj())
		else:
		env = np.einsum('xy,xsz,ysw->zw', env, M, M.conj())
		return float(np.real(env[0, 0]))
	\end{verbatim}
	
	\subsubsection{Two-Point Correlators}
	
	\begin{equation}
		\langle O^{(i)}_1 O^{(j)}_2 \rangle \quad (i < j)
	\end{equation}
	\begin{verbatim}
		def correlator(mps_obj, op1, op2, i, j):
		env = np.ones((1, 1))
		for k in range(mps_obj.L):
		M = mps_obj.tensors[k]
		if k == i:
		env = np.einsum('xy,xsz,st,ytw->zw', env, M, op1, M.conj())
		elif k == j:
		env = np.einsum('xy,xsz,st,ytw->zw', env, M, op2, M.conj())
		else:
		env = np.einsum('xy,xsz,ysw->zw', env, M, M.conj())
		return float(np.real(env[0, 0]))
	\end{verbatim}
	
	\subsubsection{Entanglement Entropy}
	
	Von Neumann entropy at bond $b$ (between sites $b-1$ and $b$):
	\begin{enumerate}
		\item Left-canonicalize sites $0 \dots b-1$ by sequential QR
		\item SVD the tensor at site $b$: Schmidt values $\{\lambda_\alpha\}$
		\item $S(b) = -\sum_\alpha \lambda_\alpha^2 \ln \lambda_\alpha^2$
		(after normalizing $\sum_\alpha\lambda_\alpha^2 = 1$)
	\end{enumerate}
	
	\subsubsection{String Order Parameter}
	
	The Haldane string order parameter:
	\begin{equation}
		\mathcal{O}^z(i,j) = \left\langle S^z_i\;
		\exp\!\left(i\pi \sum_{k=i+1}^{j-1} S^z_k\right)
		S^z_j \right\rangle
	\end{equation}
	Insert the string operator $e^{i\pi S^z} = \mathrm{diag}(e^{i\pi}, 1, e^{-i\pi})$
	at all intermediate sites using transfer matrix Eq.~\eqref{eq:transfer_with_op}.

	\section{MPO Construction}
	
	\subsection{Heisenberg Model}
	
	For the Heisenberg Hamiltonian
	$H = \sum_{i} [J_x S^x_i S^x_{i+1} + J_y S^y_i S^y_{i+1}
	+ J_z S^z_i S^z_{i+1}] + \sum_i h S^z_i$,
	the bulk MPO tensor ($D_W = 5$) is:
	\begin{equation}
		W^{(i)} = \begin{pmatrix}
			\Id & 0 & 0 & 0 & 0 \\
			S^+ & 0 & 0 & 0 & 0 \\
			S^- & 0 & 0 & 0 & 0 \\
			S^z & 0 & 0 & 0 & 0 \\
			hS^z & \tfrac{J_x}{2}S^- & \tfrac{J_x}{2}S^+ & J_z S^z & \Id
		\end{pmatrix}
	\end{equation}
	Left boundary: bottom row of $W$ (shape $(1, D_W, d, d)$).
	Right boundary: left column of $W$ (shape $(D_W, 1, d, d)$).
	
	\subsection{AKLT Model via SVD Decomposition}
	\label{sec:aklt_mpo}
	
	For the AKLT spin-1 chain,
	\begin{equation}
		H = \sum_{i=1}^{L-1} \left[
		\bm{S}_i \cdot \bm{S}_{i+1}
		+ \tfrac{1}{3}\left(\bm{S}_i \cdot \bm{S}_{i+1}\right)^2
		\right]
	\end{equation}
	the MPO is constructed by SVD-decomposing the two-site kernel.
	
	\subsubsection{Two-Site Kernel}
	
	Define $H_2[\sigma_1,\sigma_2;\sigma_1',\sigma_2']
	= (\bm{S}\cdot\bm{S})_{\sigma_1\sigma_2,\sigma_1'\sigma_2'}
	+ \tfrac{1}{3}(\bm{S}\cdot\bm{S})^2_{\sigma_1\sigma_2,\sigma_1'\sigma_2'}$
	with $d=3$ (spin-1).
	Reorder to $H_2^r[\sigma_1,\sigma_1',\sigma_2,\sigma_2']$ and reshape to
	a $(d^2)\times(d^2)$ matrix:
	
	\begin{equation}
		H_2^r\big|_{(\sigma_1\sigma_1'),(\sigma_2\sigma_2')}
		= \sum_{k=1}^{r} A_k[\sigma_1,\sigma_1']\; B_k[\sigma_2,\sigma_2']
	\end{equation}
	
	where $A_k, B_k$ are obtained from SVD:
	$U\,\mathrm{diag}(\sigma)\,V^\dagger$ with
	$A_k = \sqrt{\sigma_k}\,U_k.\text{reshape}(d,d)$,
	$B_k = \sqrt{\sigma_k}\,V^\dagger_k.\text{reshape}(d,d)$.
	For the AKLT model, the numerical rank is $r = 9$.
	
	\subsubsection{MPO Bond Structure}
	
	With $D_W = r+2 = 11$, the bulk $W$ matrix has the structure:
	
	\begin{center}
		\begin{tabular}{c|ccc}
			\toprule
			row $\backslash$ col & $0$ & $1\ldots r$ & $r+1$ \\
			\midrule
			$0$     & $\Id$    & $0$     & $0$  \\
			$1\ldots r$ & $B_k$ & $0$     & $0$  \\
			$r+1$   & $0$      & $A_k$   & $\Id$ \\
			\bottomrule
		\end{tabular}
	\end{center}
	
	\begin{itemize}
		\item State $0$: ``done'' — passes accumulated terms rightward
		\item States $1\ldots r$: ``active$_k$'' — carries $B_k$ to the right
		\item State $r+1$: ``waiting'' — not yet started; injects $A_k$
	\end{itemize}
	
	Boundary tensors (same convention as Heisenberg):
	\begin{align}
		W_L &= W[r+1:\,,\, :,\, :,\, :] \quad \text{shape }(1, D_W, d, d)
		\quad\text{(start in ``waiting'' state)}\\
		W_R &= W[\,:\,,\, 0:1,\, :,\, :] \quad \text{shape }(D_W, 1, d, d)
		\quad\text{(end in ``done'' state)}
	\end{align}
	
	\subsubsection{Two-Site Verification}
	
	\begin{equation}
		\sum_b W_L[0, b, \sigma_1, \sigma_1']\; W_R[b, 0, \sigma_2, \sigma_2']
		= H_2^r[\sigma_1,\sigma_1',\sigma_2,\sigma_2']
	\end{equation}
	In code:
	\begin{verbatim}
		H2_check = np.einsum('ibsS,bjtT->sStT', W_L, W_R)
		# index b: a_out of W_L contracts with a_in of W_R
		err = np.max(np.abs(H2_check - H2_r))   # should be < 1e-13
	\end{verbatim}
	
	\subsubsection{Python Implementation}
	
	\begin{verbatim}
		def build_aklt_mpo(L):
		d = 3
		# Build two-site Hamiltonian H2_4[s1,s2,s1',s2']
		SS = (np.einsum('ij,kl->ikjl', Sz, Sz)
		+ 0.5*np.einsum('ij,kl->ikjl', Sp, Sm)
		+ 0.5*np.einsum('ij,kl->ikjl', Sm, Sp))
		SS_mat = SS.reshape(d*d, d*d)
		H2_mat = SS_mat + (1.0/3.0)*SS_mat @ SS_mat
		H2_r = H2_mat.reshape(d,d,d,d).transpose(0,2,1,3) # [s1,s1',s2,s2']
		
		U, sv, Vt = np.linalg.svd(H2_r.reshape(d*d, d*d), full_matrices=False)
		r = int(np.sum(sv > 1e-12))
		sqsv = np.sqrt(sv[:r])
		A = (U[:,:r]*sqsv[None,:]).T.reshape(r,d,d)  # A_k[s1,s1']
		B = (Vt[:r,:]*sqsv[:,None]).reshape(r,d,d)   # B_k[s2,s2']
		
		D_W = r + 2
		W = np.zeros((D_W, D_W, d, d))
		W[0,   0,   :, :] = np.eye(d)        # done -> done
		W[r+1, r+1, :, :] = np.eye(d)        # waiting -> waiting
		for k in range(r):
		W[r+1, k+1, :, :] = A[k]         # waiting -> active_k
		W[k+1, 0,   :, :] = B[k]         # active_k -> done
		
		W_L = W[r+1:r+2, :, :, :]            # shape (1, D_W, d, d)
		W_R = W[:,  0:1,  :, :]              # shape (D_W, 1, d, d)
		return [W_L] + [W]*(L-2) + [W_R], D_W
	\end{verbatim}

	\section{Numerical Verification and Testing}
	
	\subsection{Unit Tests for Canonicalization}
	
	\textbf{Left-canonical form}:
	\begin{equation}
		\sum_{a_{i-1}, \sigma_i} A^{(i)}_{a_{i-1}, \sigma_i, a_i}
		\overline{A^{(i)}_{a_{i-1}, \sigma_i, a'_i}} = \delta_{a_i, a'_i}
	\end{equation}
	\begin{verbatim}
		def test_left_canonical(A):
		Dl, d, Dr = A.shape
		A_mat = A.reshape(Dl * d, Dr)
		assert np.allclose(A_mat.conj().T @ A_mat, np.eye(Dr), atol=1e-12)
	\end{verbatim}
	
	\textbf{Right-canonical form}:
	\begin{equation}
		\sum_{\sigma_i, a_i} B^{(i)}_{a_{i-1}, \sigma_i, a_i}
		\overline{B^{(i)}_{a'_{i-1}, \sigma_i, a_i}} = \delta_{a_{i-1}, a'_{i-1}}
	\end{equation}
	\begin{verbatim}
		def test_right_canonical(B):
		Dl, d, Dr = B.shape
		B_mat = B.reshape(Dl, d * Dr)
		assert np.allclose(B_mat @ B_mat.conj().T, np.eye(Dl), atol=1e-12)
	\end{verbatim}
	
	\subsection{MPO Two-Site Check}
	
	For any MPO with left boundary $W_L$ (shape $(1, D_W, d, d)$) and
	right boundary $W_R$ (shape $(D_W, 1, d, d)$):
	\begin{verbatim}
		H2_check = np.einsum('ibsS,bjtT->sStT', W_L, W_R)
		# b = a_out of W_L = a_in of W_R
		assert np.max(np.abs(H2_check - H2_exact)) < 1e-12
	\end{verbatim}
	
	\subsection{Common Numerical Issues and Fixes}
	
	\begin{enumerate}
		\item \textbf{Norm explosion/collapse}: If \texttt{mps.norm\_sq()} deviates
		from 1, the MPS was not properly canonicalized before DMRG.
		Fix: Always right-canonicalize before building environments.
		
		\item \textbf{Einsum index collision}: The most common source of silent errors.
		When writing $M^*[y,t,w]$, all three indices must be distinct from
		$M[x,s,z]$ and \texttt{env}$[x,y]$.
		The safe convention is:
		\begin{center}
			\begin{tabular}{lll}
				\toprule
				Tensor & Indices & Sizes \\
				\midrule
				\texttt{env} & $x, y$ & $D_l \times D_l$ \\
				$M$ & $x, s, z$ & $D_l \times d \times D_r$ \\
				$M^*$ & $y, t, w$ & $D_l \times d \times D_r$ \\
				$O$ (operator) & $s, t$ & $d \times d$ \\
				output & $z, w$ & $D_r \times D_r$ \\
				\bottomrule
			\end{tabular}
		\end{center}
		
		\item \textbf{Energy off by factor of 2}: Indicates MPO bond indices
		contracted in wrong direction (both $a_{\rm in}$ instead of
		$a_{\rm out}$ of one contracting with $a_{\rm in}$ of next).
		Check the two-site MPO verification einsum.
		
		\item \textbf{Environment index mismatch}: \texttt{envR} output index
		appears in output but never in inputs (NumPy \texttt{ValueError}).
		Ensure all output indices of the environment einsum appear
		in at least one input tensor.
	\end{enumerate}
	
	\section{Performance Optimization}
	
	\subsection{Computational Complexity}
	
	For two-site DMRG with bond dimension $D$, MPO bond dimension $D_W$,
	physical dimension $d$:
	
	\begin{itemize}
		\item \textbf{Environment update}: $\Order(D_W D^2 d^2)$ per site
		\item \textbf{Effective Hamiltonian matvec}: $\Order(D_W D^2 d^2)$
		(two-step contraction)
		\item \textbf{Lanczos/\texttt{eigsh} iteration}: $\Order(n_{\rm iter}
		\cdot D_W D^2 d^2)$, $n_{\rm iter} \sim 10$--$50$
		\item \textbf{Total per sweep}: $\Order(L \cdot n_{\rm iter}
		\cdot D_W D^2 d^2)$
	\end{itemize}
	
	\textit{Memory}: $\Order(L D^2 d + L D_W D^2)$ for MPS, MPO, and environments.
	
	\subsection{Optimization Strategies}
	
	\begin{enumerate}
		\item \textbf{Two-step matvec}: Never form the full $\mathcal{H}_{\rm eff}$
		matrix; always use the two-step contraction
		(Eqs.~\eqref{eq:matvec_step1}--\eqref{eq:matvec_step2})
		to keep cost at $\Order(D_W D^2 d^2)$ instead of $\Order(D^4 d^2)$.
		
		\item \textbf{Warm start}: Initialize \texttt{eigsh} with the previous
		optimized $\Theta_0$; reduces Lanczos iterations by $\sim 5\times$.
		
		\item \textbf{Fallback to dense diagonalization}: For small bond dimensions
		($D_l d^2 D_r < 500$), dense \texttt{np.linalg.eigh} is often faster
		than \texttt{eigsh} due to setup overhead.
		
		\item \textbf{Adaptive bond dimension}: Start with small $D$, increase
		as needed. For AKLT: $D=2$ already exact; $D=5$ confirms zero error.
	\end{enumerate}

	\section{Extensions and Advanced Topics}
	
	\subsection{Finite Temperature (DMRG-FT)}
	
	Purification approach: represent $\rho = e^{-\beta H}/Z$ as a pure state
	in doubled Hilbert space:
	\begin{equation}
		|\psi(\beta)\rangle = \frac{1}{\sqrt{Z}}
		\sum_{\{\sigma\}} e^{-\beta E_{\{\sigma\}}/2}
		|\sigma\rangle \otimes |\sigma\rangle
	\end{equation}
	
	\subsection{Time Evolution (TEBD/TDVP)}
	
	\textbf{TEBD}:
	$e^{-iH\delta t} \approx
	\prod_{\rm odd\,}i e^{-ih_{i,i+1}\delta t}
	\prod_{\rm even\,}i e^{-ih_{i,i+1}\delta t}$,
	followed by SVD truncation after each gate.
	
	\textbf{TDVP}: $i\frac{d}{dt}M^{(i)} = P_{\rm tangent}(\mathcal{H}_{\rm eff}M^{(i)})$.
	
	\subsection{Excited States}
	
	\textbf{Penalty method}:
	$\tilde{H} = H + \omega\sum_{n=0}^{N-1}|\psi_n\rangle\langle\psi_n|$
	with previously found states $|\psi_n\rangle$ and $\omega \gg \|H\|$.
	
	\subsection{Infinite MPS (iMPS)}
	
	For translation-invariant systems with unit cell of $n$ sites:
	\begin{equation}
		|\psi\rangle = \sum_{\{\sigma\}}
		\Tr\!\left(A^{[\sigma_1]} A^{[\sigma_2]} \cdots A^{[\sigma_n]}
		A^{[\sigma_1]} \cdots\right)
		|\sigma_1\sigma_2\cdots\rangle
	\end{equation}
	Fixed point: $\rho_L = \sum_\sigma A^{[\sigma]}\rho_L A^{[\sigma]\dagger}$.

	Before implementing, ensure understanding of:
	
	\begin{enumerate}
		\item[$\square$] Index conventions: $W[a_{\rm in}, a_{\rm out}, \sigma, \sigma']$
		\item[$\square$] Canonical forms and their verification
		\item[$\square$] Environment tensor shapes and update einsums
		\item[$\square$] Two-site effective Hamiltonian (two-step contraction)
		\item[$\square$] \texttt{eigsh} with warm start and dense fallback
		\item[$\square$] SVD truncation with dual tolerances
		\item[$\square$] MPO construction: Heisenberg (analytical) or AKLT (SVD)
		\item[$\square$] Two-site MPO verification einsum \texttt{'ibsS,bjtT->sStT'}
		\item[$\square$] Transfer matrix einsums: distinct indices for $M$ and $M^*$
		\item[$\square$] Entanglement entropy edge drop under OBC is physical
	\end{enumerate}
	
	\textbf{Recommended implementation order}:
	\begin{enumerate}

		\item MPO construction with two-site verification
		\item Environment updates (test with identity MPO: energy = 0)

		\item Observables: $\langle S^z_i\rangle$, correlators, entropy, string order
	\end{enumerate}

site from the rest of the chain, and the boundary spin-$1/2$ edge mode
under OBC is nearly a product state with the bulk.
Since the exact MPS bond dimension is 2, the effective Schmidt rank
at the boundary is 1, giving $S(19)\to 0$.

\subsection{String Order Parameter}

\begin{figure}[h]
	\centering
	\includegraphics[width=0.65\textwidth]{figureAKLT/06_string_order.png}
	\caption{Haldane string order parameter $\mathcal{O}^z(i_0, i_0+r)$
		as a function of distance $r$, measured from reference site $i_0=5$.
		The numerical result (green) is flat and distance-independent at
		$-4/9 = -0.4\overline{4}$ (red dashed line), confirming long-range
		hidden topological order.
		The $y$-axis offset notation $10^{-10} - 4.444\ldots\times10^{-1}$
		indicates agreement with the exact value to sub-pico precision.}
	\label{fig:string}
\end{figure}

The Haldane string order parameter
\begin{equation}
	\mathcal{O}^z(i_0, i_0+r)
	= \left\langle S^z_{i_0}\;
	\exp\!\left(i\pi\sum_{k=i_0+1}^{i_0+r-1} S^z_k\right)
	S^z_{i_0+r}\right\rangle
\end{equation}
is measured from reference site $i_0 = 5$ for distances $r = 2, \ldots, 9$
(Fig.~\ref{fig:string}).
The numerical result is flat and distance-independent:
\begin{equation}
	\mathcal{O}^z \approx -4.444\ldots\times10^{-1}
	= -\frac{4}{9} \quad \text{(exact)},
\end{equation}
with deviation $< 10^{-10}$.
The long-range nature of the string order, constant for all $r$,
is the defining signature of the Haldane phase and confirms that the
DMRG ground state correctly captures the hidden topological order.
